## 4.1 Authentication Routes

```javascript
// routes/auth.js
const express = require('express');
const router = express.Router();
const bcrypt = require('bcrypt');
const jwt = require('jsonwebtoken');
const pool = require('../config/database');

// POST /api/auth/signup
router.post('/signup', async (req, res) => {
  try {
    const { email, password, username } = req.body;
    
    // Validation
    if (!email || !password) {
      return res.status(400).json({ error: 'Email and password required' });
    }
    
    // Check if user exists
    const existingUser = await pool.query(
      'SELECT * FROM users WHERE email = $1',
      [email]
    );
    
    if (existingUser.rows.length > 0) {
      return res.status(409).json({ error: 'User already exists' });
    }
    
    // Hash password
    const hashedPassword = await bcrypt.hash(password, 10);
    
    // Create user
    const newUser = await pool.query(
      'INSERT INTO users (email, password_hash, username) VALUES ($1, $2, $3) RETURNING id, email, username',
      [email, hashedPassword, username || email.split('@')[0]]
    );
    
    // Generate JWT token
    const token = jwt.sign(
      { userId: newUser.rows[0].id, email: newUser.rows[0].email },
      process.env.JWT_SECRET,
      { expiresIn: '7d' }
    );
    
    res.status(201).json({
      success: true,
      user: newUser.rows[0],
      token
    });
  } catch (error) {
    console.error('Signup error:', error);
    res.status(500).json({ error: 'Signup failed' });
  }
});

// POST /api/auth/login
router.post('/login', async (req, res) => {
  try {
    const { email, password } = req.body;
    
    if (!email || !password) {
      return res.status(400).json({ error: 'Email and password required' });
    }
    
    // Find user
    const user = await pool.query(
      'SELECT * FROM users WHERE email = $1',
      [email]
    );
    
    if (user.rows.length === 0) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }
    
    // Verify password
    const validPassword = await bcrypt.compare(password, user.rows[0].password_hash);
    
    if (!validPassword) {
      return res.status(401).json({ error: 'Invalid credentials' });
    }
    
    // Generate JWT token
    const token = jwt.sign(
      { userId: user.rows[0].id, email: user.rows[0].email },
      process.env.JWT_SECRET,
      { expiresIn: '7d' }
    );
    
    res.json({
      success: true,
      user: {
        id: user.rows[0].id,
        email: user.rows[0].email,
        username: user.rows[0].username
      },
      token
    });
  } catch (error) {
    console.error('Login error:', error);
    res.status(500).json({ error: 'Login failed' });
  }
});

// POST /api/auth/logout
router.post('/logout', (req, res) => {
  // JWT is stateless, logout just needs to clear token on client
  res.json({ success: true, message: 'Logged out successfully' });
});

module.exports = router;
```

```javascript
const express = require('express');
const router = express.Router();

       RETURNING *`,
      [req.userId, content_type, niche, title, description, image_url, video_url, affiliate_link, 'draft', scheduled_dates || [], tags || []]
    );
    
    res.status(201).json({

    res.status(500).json({ error: 'Failed to create content' });

// GET /api/content/list
router.get('/list', verifyAuth, async (req, res) => {
  try {
    const { niche, status, limit = 20, offset = 0 } = req.query;
    
    let query = 'SELECT * FROM content_library WHERE user_id = $1';
    const params = [req.userId];
    
    if (niche) {
      query += ` AND niche = $${params.length + 1}`;
      params.push(niche);
    }
    
    if (status) {
      query += ` AND status = $${params.length + 1}`;
      params.push(status);
    }
    
    query += ` ORDER BY created_at DESC LIMIT $${params.length + 1} OFFSET $${params.length + 2}`;
    params.push(limit, offset);
    
    const result = await pool.query(query, params);
    
    // Get total count
    let countQuery = 'SELECT COUNT(*) FROM content_library WHERE user_id = $1';
    const countParams = [req.userId];
    
    if (niche) {
      countQuery += ` AND niche = $${countParams.length + 1}`;
      countParams.push(niche);
    }
    
    if (status) {
      countQuery += ` AND status = $${countParams.length + 1}`;
      countParams.push(status);
    }
    
    const countResult = await pool.query(countQuery, countParams);
    
    res.json({
      success: true,
      content: result.rows,
      total: parseInt(countResult.rows[0].count),
      limit: parseInt(limit),
      offset: parseInt(offset)
    });
  } catch (error) {
    console.error('List content error:', error);
    res.status(500).json({ error: 'Failed to list content' });
  }

  try {
    const { content_id, platforms, schedule } = req.body;
    
    // Get content
    const contentResult = await pool.query(
      'SELECT * FROM content_library WHERE id = $1 AND user_id = $2',
      [content_id, req.userId]
    );
    
    if (contentResult.rows.length === 0) {
      return res.status(404).json({ error: 'Content not found' });
    }
    
    const content = contentResult.rows[0];
    
    // Create scheduled posts for each platform and date
    const scheduledPosts = [];
    
    for (const platform of platforms) {
      for (const { date, time } of schedule) {
        const scheduledTime = new Date(`${date}T${time}:00Z`);

          `INSERT INTO scheduled_posts (user_id, content_id, platform, scheduled_time, execution_status)
           VALUES ($1, $2, $3, $4, 'pending')
           RETURNING *`,
          [req.userId, content_id, platform, scheduledTime]
        );
        
        scheduledPosts.push(result.rows[0]);
      }
    }
    
    // Update content status to scheduled
    await pool.query(
      'UPDATE content_library SET status = $1 WHERE id = $2',
      ['scheduled', content_id]
    );
    
    res.json({
      success: true,
      scheduledPosts
    });
  } catch (error) {
    console.error('Schedule content error:', error);
    res.status(500).json({ error: 'Failed to schedule content' });
  }
});

module.exports = router;
```

```javascript
// routes/analytics.js
const express = require('express');
const router = express.Router();
const pool = require('../config/database');
const { verifyAuth } = require('../middleware/auth');

// GET /api/analytics/dashboard
router.get('/dashboard', verifyAuth, async (req, res) => {
  try {
    const { date_range = '30days' } = req.query;
    
    // Calculate date range
    const daysAgo = date_range === '7days' ? 7 : date_range === '90days' ? 90 : 30;
    const startDate = new Date();
    startDate.setDate(startDate.getDate() - daysAgo);
    
    // Get revenue data
    const revenueResult = await pool.query(
      `SELECT 
        SUM(metric_value) as total_revenue,
        COUNT(*) FILTER (WHERE event_type = 'link_clicked') as total_clicks,
        COUNT(*) FILTER (WHERE event_type = 'conversion') as total_conversions
       FROM analytics_events
       WHERE user_id = $1 AND created_at >= $2`,
      [req.userId, startDate]
    );
    
    const revenue = revenueResult.rows[0];
    
    // Get revenue by niche
    const nicheResult = await pool.query(
      `SELECT 
        cl.niche,
        SUM(ae.metric_value) as revenue,
        COUNT(*) FILTER (WHERE ae.event_type = 'link_clicked') as clicks,
        COUNT(*) FILTER (WHERE ae.event_type = 'conversion') as conversions
       FROM analytics_events ae
       LEFT JOIN content_library cl ON ae.content_id = cl.id
       WHERE ae.user_id = $1 AND ae.created_at >= $2
       GROUP BY cl.niche
       ORDER BY revenue DESC`,
      [req.userId, startDate]
    );
    
    // Get revenue by platform
    const platformResult = await pool.query(
      `SELECT 
        platform,
        SUM(metric_value) as revenue,
        COUNT(*) FILTER (WHERE event_type = 'link_clicked') as clicks,
        COUNT(*) FILTER (WHERE event_type = 'conversion') as conversions
       FROM analytics_events
       WHERE user_id = $1 AND created_at >= $2
       GROUP BY platform
       ORDER BY revenue DESC`,
      [req.userId, startDate]
    );
    
    // Get top performers
    const topContentResult = await pool.query(
      `SELECT 
        cl.id,
        cl.title,
        cl.content_type,
        cl.performance_metrics,
        COUNT(*) as clicks,
        COUNT(*) FILTER (WHERE ae.event_type = 'conversion') as conversions
       FROM content_library cl
       LEFT JOIN analytics_events ae ON cl.id = ae.content_id
       WHERE cl.user_id = $1 AND ae.created_at >= $2
       GROUP BY cl.id
       ORDER BY clicks DESC
       LIMIT 5`,
      [req.userId, startDate]
    );
    
    // Get trends (daily revenue)
    const trendsResult = await pool.query(
      `SELECT 
        DATE(created_at) as date,
        SUM(metric_value) as revenue,
        COUNT(*) as events
       FROM analytics_events
       WHERE user_id = $1 AND created_at >= $2
       GROUP BY DATE(created_at)
       ORDER BY date ASC`,
      [req.userId, startDate]
```

# PART 6: AUTHENTICATION MIDDLEWARE

```javascript
// middleware/auth.js
const jwt = require('jsonwebtoken');

const verifyAuth = (req, res, next) => {
  try {
    const authHeader = req.headers.authorization;
    
    if (!authHeader) {
      return res.status(401).json({ error: 'No authorization header' });
    }
    
    const token = authHeader.replace('Bearer ', '');
    
    const decoded = jwt.verify(token, process.env.JWT_SECRET);
    
    req.userId = decoded.userId;
    req.email = decoded.email;
    
    next();
  } catch (error) {
    console.error('Auth error:', error);
    res.status(401).json({ error: 'Invalid or expired token' });
  }
};

module.exports = { verifyAuth };
```

```javascript
// jobs/dailyPostingJob.js
const cron = require('node-cron');
const pool = require('../config/database');
const bufferService = require('../services/bufferService');
const pinterestService = require('../services/pinterestService');

// Run every day at 6 AM UTC
cron.schedule('0 6 * * *', async () => {
  console.log('⏰ Daily posting job started');
  
  try {
    // Get all pending scheduled posts for today
    const result = await pool.query(`
      SELECT sp.*, cl.title, cl.image_url, cl.description, cl.affiliate_link
      FROM scheduled_posts sp
      JOIN content_library cl ON sp.content_id = cl.id
      WHERE DATE(sp.scheduled_time) = CURRENT_DATE
      AND sp.execution_status = 'pending'
    `);
    
    for (const post of result.rows) {
      try {
        // Execute based on platform
        if (post.platform === 'pinterest') {
          await pinterestService.createPin(
            'board-id',
            post.image_url,
            post.description,
            post.affiliate_link
          );
        } else if (post.platform === 'buffer') {
          await bufferService.schedulePost(
            'profile-id',
            post.description,
            post.image_url,
            post.affiliate_link,
            post.scheduled_time
          );
        }
        
        // Update status
        await pool.query(
          'UPDATE scheduled_posts SET execution_status = $1, executed_at = NOW() WHERE id = $2',
          ['success', post.id]
        );
        
        console.log(`✅ Posted: ${post.title}`);
      } catch (error) {
        console.error(`❌ Failed to post ${post.id}:`, error);
        
        await pool.query(
          'UPDATE scheduled_posts SET execution_status = $1, error_message = $2, retry_count = retry_count + 1 WHERE id = $3',
          ['failed', error.message, post.id]
        );
      }
    }
    
    console.log('✅ Daily posting job completed');
  } catch (error) {
    console.error('❌ Daily posting job failed:', error);
  }
});
```

```javascript
// jobs/metricsSyncJob.js
const cron = require('node-cron');
const pool = require('../config/database');
```
